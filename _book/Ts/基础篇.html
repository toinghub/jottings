  abstract getNmae(): string  //抽象成员 不能被实现

  printName() {
    console.log(this.getNmae)
  }
}

class Derived extends Base {
  getNmae() {
    return 'world'
  }
}

const b = new Derived()
b.getNmae()
b.printName()
