      immediate:true,    
      // immediate表示在watch中首次绑定的时候，是否执行handler，

无刷更新url

export const replaceUrl = (hash: string) => {
    history.replaceState(
        {
            back: history.state.back,
            current: hash
        },
        "",
        `${location.search}#${hash}`
    );
}
